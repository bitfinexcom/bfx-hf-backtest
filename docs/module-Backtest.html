(async, static) execOnline(strategies, args) → {Promise}

Source:
  • exec_online.js, line 54

Executes a backtest for the specified period/symbol/tf, logs results to the console.

Example
const HFBT = require('bfx-hf-backtest')
const EMAStrategy = require('bfx-hf-strategy/examples/ema_cross')
const { SYMBOLS, TIME_FRAMES } = require('bfx-hf-util')

const now = Date.now()
const market = {
  symbol: SYMBOLS.XMR_USD,
  tf: TIME_FRAMES.ONE_MINUTE
}

const strategy = EMAStrategy(market)

await HFBT.execOnline([strategy], {
  exchange: 'bitfinex',
  from: now - (2 * 24 * 60 * 60 * 1000),
  to: now,
  trades: true,
  candles: true,
  ...market
})
